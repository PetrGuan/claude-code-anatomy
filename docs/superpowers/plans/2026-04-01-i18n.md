**Goal:** Add English/Chinese bilingual support with URL prefix routing (English default at `/`, Chinese at `/zh/`).

**Architecture:** Astro 5 built-in i18n routing with `prefixDefaultLocale: false`. Translations stored in a central TypeScript file. Each page extracted into a shared content component that accepts a `locale` prop, with thin page wrappers per locale. All hardcoded Chinese text replaced with `t(locale, key)` lookups.

**Tech Stack:** Astro 5 i18n routing, TypeScript

- [ ] **Step 1: Create locales.ts**

Write `src/i18n/locales.ts`:

```ts
export type Locale = "en" | "zh";

export const defaultLocale: Locale = "en";
export const locales: Locale[] = ["en", "zh"];

const BASE = "/claude-code-anatomy";

export function getLocalePath(locale: Locale, path: string): string {
  const cleanPath = path.startsWith("/") ? path : `/${path}`;
  if (locale === defaultLocale) return `${BASE}${cleanPath}`;
  return `${BASE}/${locale}${cleanPath}`;
}

export function getAlternatePath(currentLocale: Locale, currentPath: string): string {
  const base = BASE;
  const stripped = currentPath.replace(base, "");
  const withoutLocale = stripped.replace(/^\/(en|zh)/, "");
  const path = withoutLocale || "/";
  const targetLocale = currentLocale === "en" ? "zh" : "en";
  return getLocalePath(targetLocale, path);
}

export function getLocaleFromPath(path: string): Locale {
  const stripped = path.replace(BASE, "");
  if (stripped.startsWith("/zh")) return "zh";
  return "en";
}
```

export function t(locale: Locale, path: string): string {
  const keys = path.split(".");
  let current: any = translations;
  for (const key of keys) {
    current = current?.[key];
    if (!current) return path; // fallback: return the key itself
  }
  return (current as TranslationValue)[locale] ?? path;
}
